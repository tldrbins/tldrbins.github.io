Connect to VPN

Tools

sudo apt install strongswan

Settings

# Edit /etc/ipsec.secrets
%any : PSK "password"

# Edit /etc/ipsec.conf (copy from ike-scan result)
config setup
    charondebug="all"
    uniqueids=yes
    strictcrlpolicy=no

conn testvpn
    authby=secret
    auto=add
    ike=3des-sha1-modp1024!
    esp=3des-sha1!
    type=transport
    keyexchange=ikev1
    left=10.10.14.10
    right=10.10.11.10
    rightsubnet=10.10.11.10[tcp]

Connect

# Reset
ipsec restart

# Connect
ipsec up testvpn
