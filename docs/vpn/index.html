Tools

Settings

# Edit /etc/ipsec.secrets
%any : PSK <PASSWORD>

# Edit /etc/ipsec.conf (copy from ike-scan result)
config setup
    charondebug="all"
    uniqueids=yes
    strictcrlpolicy=no

conn testvpn
    authby=secret
    auto=add
    ike=3des-sha1-modp1024!
    esp=3des-sha1!
    type=transport
    keyexchange=ikev1
    left=<LOCAL>
    right=<TARGET>
    rightsubnet=<TARGET>[tcp]

Connect

# Reset
ipsec restart

# Connect
ipsec up testvpn
