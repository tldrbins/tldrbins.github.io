Remote Procedure Call

Tools

# Connect without creds
rpcclient -U '' -N <TARGET>

# Connect with creds
rpcclient -U '<DOMAIN>/<USER>%<PASSWORD>' <TARGET>

# Inline execute command
rpcclient -U '<DOMAIN>/<USER>%<PASSWORD>' <TARGET> -c 'querydispinfo'

Basic Commands

# General info
querydispinfo

# List of users
enumdomusers

# List of groups
enumdomgroups

# Query group by rid
querygroup <RID>

# Query group member by rid
querygroupmem <RID>

# Query user by rid
queryuser <RID>

# Look up a user
lookupnames <USER>

# Look up by SID
lookupsids <SID>

impacket-lookupsid '<USER>:<PASSWORD>@<TARGET>'
