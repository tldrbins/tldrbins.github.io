rpcclient

# Connect without creds
rpcclient -U '' -N 10.10.11.10

# Connect with creds
rpcclient -U <DOMAIN>/<USER>%<PASSWORD> 10.10.11.10

# Inline execute command
rpcclient -U <DOMAIN>/<USER>%<PASSWORD> 10.10.11.10 -c 'querydispinfo'

Basic Commands

# General info
querydispinfo

# List of users
enumdomusers

# List of groups
enumdomgroups

# Query group by rid
querygroup <RID>

# Query group member by rid
querygroupmem <RID>

# Query user by rid
queryuser <RID>

# Look up a user
lookupnames <USER>

# Look up by SID
lookupsids <SID>
