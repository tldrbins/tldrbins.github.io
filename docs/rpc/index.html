RPC

rpcclient

# Connect without creds
rpcclient -U '' -N 10.10.11.10

# Connect with creds
rpcclient -U domain/username%password 10.10.11.10

# Inline execute command
rpcclient -U domain/username%password 10.10.11.10 -c 'querydispinfo'
