zip

# zip a file
zip file.zip file

# zip a folder
zip -r folder.zip folder/

# unzip
unzip file.zip

Note: Always put the zip file inside a new folder before unzip

tar.gz

# Compress a file
tar -cvzf file.tar.gz file

# Compress a folder
tar -cvzf folder.tar.gz folder/

# Extract
tar -xvzf file.tar.gz

Note: Always put the zip file inside a new folder before Extract

gz

# Compress file, and keep the original file
gzip -k -r file

# Compress all files individually inside a folder, and keep original files
gzip -k -r folder/

# Extract
gunzip file.gz

Note: Always put the zip file inside a new folder before unzip
Note: If you want to zip the entire folder, use tar instead
