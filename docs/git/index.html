Basic Commands

# Show status
git status

# Reset the current branch to a previous commit
git reset --hard

# Show information about files
git ls-files --stage

# Show current branch
git branch

# Show commits
git log <BRANCH_NAME> --oneline

# Show diff, a and b are commit hash
# a is the older commit
git diff [a] [b]

# Show commit
git show <COMMIT_HASH>
