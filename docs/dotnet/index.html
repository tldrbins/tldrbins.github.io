Compile C# project

# Check
dotnet --info

# Prepare
mkdir <PROJECT_NAME>
cd <PROJECT_NAME>

# Init
dotnet new console

# Create .sln file
dotnet new sln

# Link .sln to .csproj
dotnet sln <PROJECT_NAME>.sln add <PROJECT_NAME>.csproj 

# Add dependencies (optional)
dotnet add package System.XXXX.XXXX.XXXX --version 1.0.0

|Add Code to Program.cs|
+----------------------+

# Build solution
dotnet build -c Release

# Run
dotnet run
