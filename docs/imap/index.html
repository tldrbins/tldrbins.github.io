IMAP/IMAPS

# Connect to IMAP over ssl
rlwrap openssl s_client -connect 10.10.11.10:993

# Login
a LOGIN username password

# List all mailboxes
a LIST "" "*"

# Select a mailbox (e.g. Inbox)
a SELECT INBOX

# Get mail from mailbox (e.g. #1)
a FETCH 1 BODY.PEEK[]
