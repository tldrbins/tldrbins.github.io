Use ping to scan subnet

# 255.255.255.0 or /24
for i in $(seq 1 254); do (ping -c 1 <SUBNET>.${i} | grep "bytes from" &); done;

Sample Output:
TO-DO
# 255.255.0.0 or /16
for i in $(seq 1 254); do for j in $(seq 1 254); do (ping -c 1 <SUBNET>.${i}.${j} | grep "bytes from" &); done; done;

Sample Output:
