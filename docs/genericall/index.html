$password.ToCharArray() | ForEach-Object {$secstr.AppendChar($_)}

# Add user to group (e.g. Exchange Windows Permissions)
Add-DomainGroupMember -Identity 'Exchange Windows Permissions' -Members USER -Credential $Cred
