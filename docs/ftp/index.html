Config Location

/etc/vsftpd.conf

Basic

+---------------------+
| Anonymous login     |
+---------------------+
| Username: anonymous |
| Password: (empty)   |
+---------------------+


# Connect
ftp <TARGET>

# Over SSL
lftp <TARGET>

# Fix cert error
echo -n 'set ssl:verify-certificate no' >> ~/.lftp/rc

# List
ls

# Switch to binary transfer mode
bin

# Download
get <FILE>

# Upload
put <FILE>

# Connect over tunnel
passive

# Exit
quit

Recusive download

# Anonymous login
wget -r ftp://anonymous:@<TARGET>

# With creds
wget --user <USER> --password '<PASSWORD>' -m ftp://<TARGET>

Note: Always check what’s in there first
